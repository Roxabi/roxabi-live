### T1 — Create `apps/web/src/lib/profile.ts` [P]

**Phase:** RED · **Agent:** frontend-dev · **Difficulty:** 2 · **SC:** SC-1, SC-4

**File:** `apps/web/src/lib/profile.ts` (new)

Model after `lib/apiKeys.ts`: fetch wrapper, `credentials: 'include'`, throw on `!res.ok`.

```ts
import type { UpdateProfilePayload, UserProfile } from '@repo/types'
import { isErrorWithMessage } from '@/lib/errorUtils'

/**
 * Fetch the current user's profile from the API.
 * Throws on network error or non-ok response.
 */
export async function getProfile(): Promise<UserProfile> {
  const res = await fetch('/api/users/me', { credentials: 'include' })
  if (!res.ok) throw new Error('fetch failed')
  return res.json() as Promise<UserProfile>
}

/**
 * Update the current user's profile.
 *
 * On API error (non-ok response): throws Error with `message` from the response body
 * (empty string if body has no message). Component distinguishes this from network errors
 * via the `isApiError` property.
 *
 * On network error: throws a generic Error (no `isApiError` property).
 */
export async function updateProfile(payload: UpdateProfilePayload): Promise<void> {
  const res = await fetch('/api/users/me', {
    method: 'PATCH',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload),
  })
  if (!res.ok) {
    const data: unknown = await res.json().catch(() => null)
    const err = Object.assign(
      new Error(isErrorWithMessage(data) ? data.message : ''),
      { isApiError: true as const }
    )
    throw err
  }
}
```

### T2 — Update `profile.lazy.tsx` to use `lib/profile.ts` [P]

**Phase:** GREEN · **Agent:** frontend-dev · **Difficulty:** 2 · **SC:** SC-2, SC-3

**File:** `apps/web/src/routes/settings/profile.lazy.tsx`

**Changes:**

1. Add import at top of file:
   ```ts
   import { getProfile, updateProfile } from '@/lib/profile'
   ```

2. Remove the `async function fetchProfile(...)` definition entirely (lines ~99–104).

3. In `useProfileData` hook, replace the `fetchProfile(...)` call with `getProfile()`, then call
   `parseProfileResponse()` in the `.then()`:
   ```ts
   // BEFORE:
   fetchProfile(user.name ?? '', user.id)
     .then((p) => set({ ...p, loaded: true }))
     .catch(() => set({ fullName: user.name ?? '', avatarSeed: user.id, loaded: true }))

   // AFTER:
   getProfile()
     .then((data) => {
       const p = parseProfileResponse(data, user.name ?? '', user.id)
       set({ ...p, loaded: true })
     })
     .catch(() => set({ fullName: user.name ?? '', avatarSeed: user.id, loaded: true }))
   ```

4. In `handleSave`, replace the inline `fetch('/api/users/me', { method: 'PATCH', ... })` block
   with a call to `updateProfile(payload)`. Distinguish API errors (thrown by `updateProfile`
   with `isApiError: true`) from network errors:
   ```ts
   async function handleSave(e: React.FormEvent) {
     e.preventDefault()
     actions.setSaving(true)
     try {
       await updateProfile({
         firstName: state.firstName || undefined,
         lastName: state.lastName || undefined,
         fullName: state.fullName,
         avatarSeed: state.avatarSeed || undefined,
         avatarStyle: state.avatarStyle,
         avatarOptions: state.avatarOptions,
         image: cdnUrl,
       })
       try {
         await authClient.updateUser({ image: cdnUrl })
       } catch {
         // Session update failed — avatar will update on next page load
       }
       toast.success(m.avatar_save_success())
     } catch (err) {
       const isApiErr = err instanceof Error && (err as { isApiError?: true }).isApiError
       const message = isApiErr && err.message ? err.message : null
       toast.error(message ?? m.avatar_save_error())
     } finally {
       actions.setSaving(false)
     }
   }
   ```

   This preserves existing test expectations:
   - API error with `{ message: 'Validation failed' }` → `toast.error('Validation failed')` ✅
   - API error with no message → `toast.error(m.avatar_save_error())` ✅
   - Network error (fetch throws) → `toast.error(m.avatar_save_error())` ✅
